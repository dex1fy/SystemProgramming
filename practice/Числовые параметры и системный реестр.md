## Запись и считывание числовых параметров в системный реестр

Шаг 1. Для выполнения задания необходимо написать следующий код в main.c

``` C
#include <Windows.h>
#include <stdio.h>

int main() {
	system("chcp 1251");

	HKEY hKey;
	if (RegOpenKeyA(HKEY_CURRENT_USER, 0, &hKey) == 0) {
		DWORD n = 30;
		if (RegSetValueExA(hKey, 0, 0, REG_DWORD, &n, sizeof(DWORD)) == 0) {
			DWORD type;
			DWORD data;
			DWORD len;
			if (RegGetValueA(hKey, 0, 0, RRF_RT_REG_DWORD, &type, &data, &len) == 0) {
				printf("%d\n", data);
			}
		}
	}
}
```
