## Программа для преобразования строки в число. Строка хранится в виде указателя LPWSTR. Реализовать без использования системных функций

Шаг 1. Для этого задания в файле main.c напишем следующий код

``` C
#include <Windows.h>
#include <stdio.h>

// преобразование строки WINAPI в число
int LPWSTR_to_int(LPWSTR str) {
	int result = 0;
	int sign = 1;
	int i = 0;

	// убрал все пробелы в самомначале
	while (str[i] == L' ') {
		i++;
	}

	// если есть - то число отрицательное
	if (str[i] == L'-') {
		sign = -1;
		i++;
	}

	// пока у нас числа мы конвертируем
	while (str[i] >= L'0' && str[i] <= L'9') {
		result = result * 10 + (str[i] - L'0');
		i++;
	}

	// возвращаем число + меняем знак
	return result * sign;
}

int main() {
	system("chcp 1251");
	LPWSTR str = L"123";

	int num = LPWSTR_to_int(str);
	printf("%d", num);
}
```
